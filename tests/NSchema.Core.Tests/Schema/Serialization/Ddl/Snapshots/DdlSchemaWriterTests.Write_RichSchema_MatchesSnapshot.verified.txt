--- application schema
CREATE PARTIAL SCHEMA app RENAMED FROM legacy_app;
GRANT USAGE ON SCHEMA app TO app_role;

--- order lifecycle
CREATE ENUM app.order_status ('pending', 'shipped', 'delivered');

CREATE ENUM app.priority RENAMED FROM importance ('low', 'high');

CREATE SEQUENCE app.invoice_id RENAMED FROM bill_id;

--- order numbers
CREATE SEQUENCE app.order_id (AS bigint, START 100, INCREMENT 5, MINVALUE -10, MAXVALUE 999999, CACHE 10, CYCLE);

--- all users
CREATE TABLE app.users RENAMED FROM members (
  id bigint NOT NULL IDENTITY (START 1, INCREMENT 1, MINVALUE 1),
  --- display name
  name varchar(255) NOT NULL,
  balance decimal(18,2) DEFAULT 0,
  code char(8) NOT NULL RENAMED FROM short_code,
  metadata jsonb,
  --- surrogate key
  CONSTRAINT users_pkey PRIMARY KEY (id),
  --- owning org
  CONSTRAINT users_org_fk FOREIGN KEY (org_id) REFERENCES app.orgs (id) ON DELETE CASCADE ON UPDATE SET NULL,
  --- external code
  CONSTRAINT users_code_uq UNIQUE (code),
  --- no overdraft
  CONSTRAINT users_balance_chk CHECK (balance >= 0),
  --- unique names
  UNIQUE INDEX users_name_ix (name) WHERE (name IS NOT NULL)
);
GRANT SELECT, INSERT, UPDATE, DELETE ON app.users TO readers;

--- currently active users
CREATE VIEW app.active_users AS SELECT id, name FROM app.users WHERE balance > 0;

CREATE VIEW app.user_directory RENAMED FROM legacy_directory AS SELECT name FROM app.active_users;
DROP TABLE app.old_table;
DROP VIEW app.stale_report;
DROP ENUM app.stale_enum;
DROP SEQUENCE app.stale_seq;

DROP SCHEMA scratch;
